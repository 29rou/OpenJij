45  template <typename RandomNumberEngine>
46  static void update(system::ContinuousTimeIsing<GraphType, false>& system,
47  RandomNumberEngine& random_number_engine,
48  const utility::TransverseFieldUpdaterParameter& parameter) {
49 
50  const graph::Index num_spin = system.num_spins;
51  std::vector<graph::Index> index_helper;
52  index_helper.reserve(num_spin+1);
53  index_helper.push_back(0);
54  /* index_helper[i]+k gives 1 dimensionalized index of kth time point at ith site.
55  * this helps use of union-find tree only available for 1D structure.
56  */
57 
58  /* 1. remove old cuts and place new cuts for every site */
59  for(graph::Index i = 0;i < num_spin;i++) {
60  auto& timeline = system.spin_config[i];
61  auto cuts = generate_poisson_points(0.5*system.gamma*(1.0-parameter.s), parameter.beta, random_number_engine);
62  // assuming transverse field gamma is positive
63 
64  timeline = create_timeline(timeline, cuts);
65  assert(timeline.size() > 0);
66 
67  index_helper.push_back(index_helper.back()+timeline.size());
68  }
69 
70  /* 2. place spacial bonds */
71  utility::UnionFind union_find_tree(index_helper.back());
72  for(graph::Index i = 0;i < num_spin;i++) {
73  for(auto&& j : system.interaction.adj_nodes(i)) {
74  if (i < j) {
75  continue; // ignore duplicated interaction
76  // if adj_nodes are sorted, this "continue" can be replaced by "break"
77  }
78 
79  const auto bonds = generate_poisson_points(std::abs(0.5*system.interaction.J(i, j)*parameter.s),
80  parameter.beta, random_number_engine);
81  for(const auto bond : bonds) {
82  /* get time point indices just before the bond */
83  auto ki = system.get_temporal_spin_index(i, bond);
84  auto kj = system.get_temporal_spin_index(j, bond);
85 
86  if(system.spin_config[i][ki].second * system.spin_config[j][kj].second * system.interaction.J(i, j) < 0) {
87  union_find_tree.unite_sets(index_helper[i]+ki, index_helper[j]+kj);
88  }
89  }
90  }
91  }
92 
93  /* 3. make clusters */
94  std::unordered_map<utility::UnionFind::Node, std::vector<utility::UnionFind::Node>> cluster_map;
95  // root-index to {contained nodes} map
96 
97  for(graph::Index i = 0;i < num_spin;i++) {
98  for(size_t k = 0;k < system.spin_config[i].size();k++) {
99  auto index = index_helper[i] + k;
100  auto root_index = union_find_tree.find_set(index);
101  auto position = cluster_map.find(root_index);
102  if(position == cluster_map.end()) {
103  cluster_map.emplace(root_index, std::vector<utility::UnionFind::Node>{ index });
104  } else {
105  position->second.push_back(index);
106  }
107  }
108  }
109 
110  /* 4. flip clusters */
111  auto urd = std::uniform_real_distribution<>(0, 1.0);
112  for(const auto& cluster : cluster_map) {
113  // 4.1. decide spin state (flip with the probability 1/2)
114  const FloatType probability = 1.0 / 2.0;
115  if(urd(random_number_engine) < probability){
116  // 4.2. update spin states
117  for(auto node : cluster.second) {
118  auto i = std::distance(index_helper.begin(),
119  std::upper_bound(index_helper.begin(), index_helper.end(), node)) - 1;
120  auto k = node - index_helper[i];
121  system.spin_config[i][k].second *= -1;
122  }
123  }
124  }
125  }

131  static std::vector<CutPoint> create_timeline(const std::vector<CutPoint>& old_timeline,
132  const std::vector<TimeType>& cuts) {
133  /* remove redundant cuts*/
134  std::vector<CutPoint> concatenated_timeline;
135  auto current_spin = old_timeline.back().second;
136  for(auto cut_point : old_timeline) {
137  if(cut_point.second != current_spin) {
138  concatenated_timeline.push_back(cut_point);
139  }
140  current_spin = cut_point.second;
141  }
142 
143  /* if entire timeline is occupied by single spin state */
144  std::vector<CutPoint> new_timeline;
145  if(concatenated_timeline.empty()) {
146  if(cuts.empty()) {
147  new_timeline.push_back(old_timeline[0]);
148  } else {
149  for(auto cut : cuts) {
150  new_timeline.push_back(CutPoint(cut, current_spin));
151  }
152  }
153  return new_timeline;
154  }
155 
156  current_spin = concatenated_timeline.back().second;
157  auto timeline_itr = concatenated_timeline.begin();
158  auto cuts_itr = cuts.begin();
159  while(true) {
160  /* if all cuts have been placed, add remaining old kinks and break loop */
161  if(cuts_itr == cuts.end()) {
162  std::for_each(timeline_itr, concatenated_timeline.end(), [&](CutPoint it) {
163  new_timeline.push_back(it);
164  });
165  break;
166  }
167 
168  /* if all spin kinks have been placed, add remaining cuts and break loop */
169  if(timeline_itr == concatenated_timeline.end()) {
170  std::for_each(cuts_itr, cuts.end(), [&](TimeType it) {
171  new_timeline.push_back(CutPoint(it, current_spin));
172  });
173  break;
174  }
175 
176  /* add earlier of kink or cut to new timeline */
177  if(*cuts_itr < timeline_itr->first) {
178  new_timeline.push_back(CutPoint(*cuts_itr, current_spin));
179  cuts_itr++;
180  } else {
181  new_timeline.push_back(*timeline_itr);
182  current_spin = timeline_itr->second;
183  timeline_itr++;
184  }
185  }
186 
187  return new_timeline;
188  }

194  static std::vector<CutPoint> create_timeline_easy(const std::vector<CutPoint>& old_timeline,
195  const std::vector<TimeType>& cuts) {
196  /* remove redundant cuts*/
197  std::vector<CutPoint> new_timeline;
198  auto current_spin = old_timeline.back().second;
199  for(auto cut_point : old_timeline) {
200  if(cut_point.second != current_spin) {
201  new_timeline.push_back(cut_point);
202  }
203  current_spin = cut_point.second;
204  }
205 
206  /* if entire timeline is occupied by single spin state */
207  if(new_timeline.empty()) {
208  if(cuts.empty()) {
209  new_timeline.push_back(old_timeline[0]);
210  } else {
211  for(auto cut : cuts) {
212  new_timeline.push_back(CutPoint(cut, current_spin));
213  }
214  }
215  return new_timeline;
216  }
217 
218  static const auto first_lt = [](CutPoint x, CutPoint y) { return x.first < y.first; };
219  for(auto cut : cuts) {
220  const auto dummy_cut = CutPoint(cut, 0); // dummy cut for binary search
221  auto itr = std::upper_bound(new_timeline.begin(),
222  new_timeline.end(),
223  dummy_cut,
224  first_lt);
225  auto prev_itr = (itr == new_timeline.begin()) ? new_timeline.end() - 1 : itr - 1;
226 
227  new_timeline.insert(itr, CutPoint(cut, prev_itr->second));
228  }
229 
230  return new_timeline;
231  }

238  template<typename RandomNumberEngine>
239  static std::vector<TimeType> generate_poisson_points(const TimeType lambda, const TimeType beta,
240  RandomNumberEngine& random_number_engine) {
241  std::uniform_real_distribution<> rand(0.0, 1.0);
242  std::uniform_real_distribution<> rand_beta(0.0, beta);
243 
244  const TimeType coef = beta*lambda;
245  TimeType n = 0;
246  TimeType d = std::exp(-coef);
247  TimeType p = d;
248  TimeType xi = rand(random_number_engine);
249 
250  while(p < xi) {
251  n += 1;
252  d *= coef/n;
253  p += d;
254  }
255 
256  std::vector<TimeType> poisson_points(n);
257  for(int k = 0;k < n;k++) {
258  poisson_points[k] = rand_beta(random_number_engine);
259  }
260  std::sort(poisson_points.begin(), poisson_points.end());
261 
262  return poisson_points;
263  }
