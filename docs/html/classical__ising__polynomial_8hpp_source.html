44  ClassicalIsingPolynomial(const graph::Spins &init_spins, const graph::Polynomial<FloatType> &init_interaction):
45  num_spins(init_interaction.get_num_spins()), spin(init_spins), vartype(init_interaction.get_vartype()) {
46 
47  //Check if the number of the initial spins/binaries is equal to num_spins defined from the Polynomial graph class.
48  assert(init_spins.size() == num_spins);
49 
50  //Check if the number of interactions is larger than 0

64  const_cast<Interactions&>(init_interaction.get_interactions()) :
65  relabel_interactions(const_cast<Interactions&>(init_interaction.get_interactions()));

89  void check_variables() {
90  if (vartype == cimod::Vartype::SPIN) {
91  for (std::size_t i = 0; i < spin.size(); ++i) {
92  if (spin[i] != -1 && spin[i] != 1) {
93  std::stringstream ss;
94  ss << "The variable at " << i << " is " << spin[i] << ".\n";
95  ss << "But the spin variable must be -1 or +1.\n";
96  throw std::runtime_error(ss.str());
97  }
98  }
99  }
100  else if (vartype == cimod::Vartype::BINARY) {
101  for (std::size_t i = 0; i < spin.size(); ++i) {
102  if (spin[i] != 0 && spin[i] != 1) {
103  std::stringstream ss;
104  ss << "The variable at " << i << " is " << spin[i] << ".\n";
105  ss << "But the binary variable must be 0 or 1.\n";
106  throw std::runtime_error(ss.str());
107  }
108  }
109  }
110  else {
111  std::stringstream ss;
112  ss << "Unknown vartype detected in " << __func__ << std::endl;
113  throw std::runtime_error(ss.str());
114  }

144  void set_update_matrix(std::vector<std::vector<graph::Index>> &interacted_spins) {
145  crs_col.clear();
146  crs_row.clear();
147  crs_val.clear();
148  crs_row.push_back(0);
149  if (vartype == cimod::Vartype::SPIN) {
150  crs_sign_p.clear();
151  for (graph::Index index_spin_row = 0; index_spin_row < num_spins; ++index_spin_row) {
152  std::vector<graph::Index> temp_vec;
153  std::vector<std::vector<FloatType>> temp_vec_val(num_spins);
154  std::vector<std::vector<int8_t*>> temp_vec_sign(num_spins);
155  for (const auto &index_spin_term: connected_J_term_index_[index_spin_row]) {
156  auto temp_J_term = J_term_[index_spin_term];
157  auto temp_sign_p = &sign_[index_spin_term];
158  for (const auto &index_spin_col: interacted_spins[index_spin_term]) {
159  if (index_spin_row != index_spin_col) {
160  temp_vec.push_back(index_spin_col);
161  temp_vec_val[index_spin_col].push_back(4*temp_J_term);
162  temp_vec_sign[index_spin_col].push_back(temp_sign_p);
163  }
164  }
165  }
166  std::sort(temp_vec.begin(), temp_vec.end());
167  temp_vec.erase(std::unique(temp_vec.begin(), temp_vec.end()), temp_vec.end());
168  for (const auto &it: temp_vec) {
169  for (std::size_t i = 0; i < temp_vec_val[it].size(); ++i) {
170  crs_col.push_back(it);
171  crs_val.push_back(temp_vec_val[it][i]);
172  crs_sign_p.push_back(temp_vec_sign[it][i]);
173  }
174  }
175  crs_row.push_back(crs_col.size());
176  }
177  }
178  else if (vartype == cimod::Vartype::BINARY) {
179  crs_zero_count_p.clear();
180  for (graph::Index index_spin_row = 0; index_spin_row < num_spins; ++index_spin_row) {
181  std::vector<graph::Index> temp_vec;
182  std::vector<std::vector<FloatType>> temp_vec_val(num_spins);
183  std::vector<std::vector<uint64_t*>> temp_vec_zero_count(num_spins);
184  for (const auto &index_spin_term: connected_J_term_index_[index_spin_row]) {
185  auto temp_J_term = J_term_[index_spin_term];
186  auto temp_zero_count_p = &zero_count_[index_spin_term];
187  for (const auto &index_spin_col: interacted_spins[index_spin_term]) {
188  if (index_spin_row != index_spin_col) {
189  temp_vec.push_back(index_spin_col);
190  temp_vec_val[index_spin_col].push_back(temp_J_term);
191  temp_vec_zero_count[index_spin_col].push_back(temp_zero_count_p);
192  }
193  }
194  }
195  std::sort(temp_vec.begin(), temp_vec.end());
196  temp_vec.erase(std::unique(temp_vec.begin(), temp_vec.end()), temp_vec.end());
197  for (const auto &it: temp_vec) {
198  for (std::size_t i = 0; i < temp_vec_val[it].size(); ++i) {
199  crs_col.push_back(it);
200  crs_val.push_back(temp_vec_val[it][i]);
201  crs_zero_count_p.push_back(temp_vec_zero_count[it][i]);
202  }
203  }
204  crs_row.push_back(crs_col.size());
205  }
206  }
207  else {
208  std::stringstream ss;
209  ss << "Unknown vartype detected in " << __func__ << std::endl;
210  throw std::runtime_error(ss.str());
211  }

222  inline void update_zero_count_and_binary(const std::size_t index) {
223  if (spin[index] == 0) {
224  spin[index] = 1;
225  for (const auto &index_interaction: connected_J_term_index_[index]) {
226  zero_count_[index_interaction]--;

229  else {

231  for (const auto &index_interaction: connected_J_term_index_[index]) {

246  void update_dE_for_spin(std::size_t index) {
247  dE[index] *= -1;
248  const std::size_t begin = crs_row[index];
249  const std::size_t end = crs_row[index + 1];
250  for (std::size_t i = begin; i < end; ++i) {
251  dE[crs_col[i]] += crs_val[i]*(*crs_sign_p[i]);
252  }
253  }
254 
255  void update_dE_for_binary(std::size_t index) {
256  dE[index] *= -1;
257  const std::size_t begin = crs_row[index];
258  const std::size_t end = crs_row[index + 1];
259  for (std::size_t i = begin; i < end; ++i) {
260  graph::Index col = crs_col[i];
261  dE[col] += Sign(spin[col] + spin[index])*(crs_val[i])*zero_or_one(spin[index], spin[col], *crs_zero_count_p[i]);
262  }
263  }

265  void reset_spins(const graph::Spins& init_spins) {
266  assert(init_spins.size() == num_spins);
267  check_variables();
268 
269  if (vartype == cimod::Vartype::SPIN) {
270  for (std::size_t index = 0; index < init_spins.size(); ++index) {
271  if (spin[index] != init_spins[index]) {
272  update_dE_for_spin(index);
273  update_sign_and_spin(index);
274  }
275  }
276  }
277  else if (vartype == cimod::Vartype::BINARY) {
278  for (std::size_t index = 0; index < init_spins.size(); ++index) {
279  if (spin[index] != init_spins[index]) {
280  update_dE_for_binary(index);
281  update_zero_count_and_binary(index);
282  }
283  }
284  }
285  else {
286  std::stringstream ss;
287  ss << "Unknown vartype detected in " << __func__ << std::endl;
288  throw std::runtime_error(ss.str());
289  }
290 
291  for (std::size_t index = 0; index < init_spins.size(); ++index) {
292  assert(spin[index] == init_spins[index]);
293  }

398  void set_interacted_spins(const Interactions &interaction) {
399  interacted_spins_.clear();
400  interacted_spins_.reserve(interaction.size());
401  for (const auto &it: interaction) {
402  if (it.first.size() > 0 && std::abs(it.second) > 0.0) {
403  interacted_spins_.push_back(it.first);
404  }
405  }
406 
407  if (interacted_spins_.size() <= 0) {
408  std::stringstream ss;
409  ss << "The input interactions of non-zero value are 0." << std::endl;
410  throw std::runtime_error(ss.str());
411  }
412 
413  //nameless function to sort "interacted_spins"
414  auto comp = [](const auto &lhs, const auto &rhs) {
415  if (lhs.size() != rhs.size()) {
416  return lhs.size() < rhs.size();
417  }
418  else {
419  if (lhs[0] != rhs[0]) {
420  return lhs[0] < rhs[0];
421  }
422  else {
423  return lhs[1] < rhs[1];
424  }
425  }
426  };
427 
428  //Sort "interacted_spins" for continuous memory access
429  std::sort(interacted_spins_.begin(), interacted_spins_.end(), comp);
430  }

434  void set_J_term(Interactions &interaction) {
435  connected_J_term_index_.resize(num_spins);
436  J_term_.resize(interacted_spins_.size());
437  if (vartype == cimod::Vartype::SPIN) {
438  sign_.resize(interacted_spins_.size());
439  for (std::size_t i = 0; i < interacted_spins_.size(); ++i) {
440  graph::Spin temp_spin_multiple = 1;
441  if (interacted_spins_[i].size() > 0) {
442  for (const auto &index_spin: interacted_spins_[i]) {
443  temp_spin_multiple *= spin[index_spin];
444  connected_J_term_index_[index_spin].push_back(i);
445  }
446  }
447  J_term_[i] = interaction[interacted_spins_[i]];
448  sign_[i] = static_cast<int8_t>(temp_spin_multiple);
449  }
450  }
451  else if (vartype == cimod::Vartype::BINARY) {
452  zero_count_.resize(interacted_spins_.size());
453  for (std::size_t i = 0; i < interacted_spins_.size(); ++i) {
454  uint64_t temp_zero_count = 0;
455  graph::Binary temp_binary_multiple = 1;
456  if (interacted_spins_[i].size() > 0) {
457  for (const auto &index_binary: interacted_spins_[i]) {
458  temp_binary_multiple *= spin[index_binary];
459  connected_J_term_index_[index_binary].push_back(i);
460  if (spin[index_binary] == 0) {
461  temp_zero_count++;
462  }
463  }
464  }
465  J_term_[i] = interaction[interacted_spins_[i]];
466  zero_count_[i] = temp_zero_count;
467  }
468  }
469  else {
470  std::stringstream ss;
471  ss << "Unknown vartype detected in " << __func__ << std::endl;
472  throw std::runtime_error(ss.str());
473  }

480  Interactions relabel_interactions(Interactions &input_interactions) {
481  //Extract variables from the keys of Interactions (std::unordered_map<std::vector<graph::Index>, FloatType, utility::VectorHash>)
482  std::unordered_set<graph::Index> variable_set;
483  for (const auto &it: input_interactions) {
484  for (const auto &index: it.first) {
485  variable_set.emplace(index);
486  }
487  }
488 
489  //Convert std::unordered_set to std::vector
490  std::vector<graph::Index> variables(variable_set.begin(), variable_set.end());
491 
492  //Sort extracted variables
493  std::sort(variables.begin(), variables.end());
494 
495  //Relabel interactions
496  std::unordered_map<graph::Index, graph::Index> relabeld_variables;
497  for (std::size_t i = 0; i < variables.size(); ++i) {
498  relabeld_variables[variables[i]] = i;
499  }
500 
501  Interactions relabeled_interactions;
502 
503  for (const auto &it: input_interactions) {
504  std::vector<graph::Index> temp_vec(it.first.size());
505  for (std::size_t i = 0; i < it.first.size(); ++i) {
506  temp_vec[i] = relabeld_variables[it.first[i]];
507  }
508  relabeled_interactions[temp_vec] = it.second;
509  }
510  return relabeled_interactions;

514  void reset_dE() {
515  dE.resize(num_spins);
516  max_dE = std::abs(J_term_[0]);//Initialize
517  min_dE = std::abs(J_term_[0]);//Initialize
518  if (vartype == cimod::Vartype::SPIN) {
519  for (graph::Index i = 0; i < num_spins; ++i) {
520  FloatType temp_energy = 0.0;
521  FloatType temp_abs = 0.0;
522  bool flag = false;
523  for (const auto &it: connected_J_term_index_[i]) {
524  temp_energy += J_term_[it]*sign_[it];
525  temp_abs += std::abs(J_term_[it]);
526  }
527  dE[i] = -2*temp_energy;
528  if (flag && max_dE < 2*temp_abs) {
529  max_dE = 2*temp_abs;
530  }
531  if (flag && min_dE > 2*temp_abs) {
532  min_dE = 2*temp_abs;
533  }
534  }
535  }
536  else if (vartype == cimod::Vartype::BINARY) {
537  for (graph::Index i = 0; i < num_spins; ++i) {
538  FloatType temp_energy = 0.0;
539  FloatType temp_abs = 0.0;
540  bool flag = false;
541  auto temp_spin = spin[i];
542  for (const auto &it: connected_J_term_index_[i]) {
543  temp_energy += J_term_[it]*Sign(temp_spin)*zero_or_one(temp_spin, zero_count_[it]);
544  temp_abs += std::abs(J_term_[it]);
545  flag = true;
546  }
547  dE[i] = temp_energy;
548  if (flag && max_dE < temp_abs) {
549  max_dE = temp_abs;
550  }
551  if (flag && min_dE > temp_abs) {
552  min_dE = temp_abs;
553  }
554  }
555  }
556  else {
557  std::stringstream ss;
558  ss << "Unknown vartype detected in " << __func__ << std::endl;
559  throw std::runtime_error(ss.str());
560  }
561  }
