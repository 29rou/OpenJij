140  void reset_dE() {
141  dE.resize(num_spins);
142  max_dE_ = std::abs(poly_value_list_[0]);//Initialize
143  min_dE_ = std::abs(poly_value_list_[0]);//Initialize
144  if (vartype_ == cimod::Vartype::SPIN) {
145  for (graph::Index i = 0; i < num_spins; ++i) {
146  FloatType temp_energy = 0.0;
147  FloatType temp_abs = 0.0;
148  bool flag = false;
149  for (const auto &it: connected_J_term_index_[i]) {
150  temp_energy += poly_value_list_[it]*spin_sign_poly_[it];
151  temp_abs += std::abs(poly_value_list_[it]);
152  }
153  dE[i] = -2*temp_energy;
154  if (flag && max_dE_ < 2*temp_abs) {
155  max_dE_ = 2*temp_abs;
156  }
157  if (flag && min_dE_ > 2*temp_abs) {
158  min_dE_ = 2*temp_abs;
159  }
160  }
161  }
162  else if (vartype_ == cimod::Vartype::BINARY) {
163  for (graph::Index i = 0; i < num_spins; ++i) {
164  FloatType temp_energy = 0.0;
165  FloatType temp_abs = 0.0;
166  bool flag = false;
167  auto temp_spin = spin[i];
168  for (const auto &it: connected_J_term_index_[i]) {
169  temp_energy += poly_value_list_[it]*Sign(temp_spin)*ZeroOrOne(temp_spin, binary_zero_count_poly_[it]);
170  temp_abs += std::abs(poly_value_list_[it]);
171  flag = true;
172  }
173  dE[i] = temp_energy;
174  if (flag && max_dE_ < temp_abs) {
175  max_dE_ = temp_abs;
176  }
177  if (flag && min_dE_ > temp_abs) {
178  min_dE_ = temp_abs;
179  }
180  }
181  }
182  else {
183  std::stringstream ss;
184  ss << "Unknown vartype detected in " << __func__ << std::endl;
185  throw std::runtime_error(ss.str());
186  }

224  void update_dE_for_binary(std::size_t index) {
225  dE[index] *= -1;
226  const std::size_t begin = crs_row_[index];
227  const std::size_t end = crs_row_[index + 1];
228  for (std::size_t i = begin; i < end; ++i) {
229  graph::Index col = crs_col_[i];

235  assert(init_spins.size() == num_spins);
236  CheckInitialConditions();
237 
238  if (vartype_ == cimod::Vartype::SPIN) {
239  for (std::size_t index = 0; index < init_spins.size(); ++index) {
240  if (spin[index] != init_spins[index]) {
241  update_dE_for_spin(index);
242  update_sign_and_spin(index);
243  }
244  }
245  }
246  else if (vartype_ == cimod::Vartype::BINARY) {
247  for (std::size_t index = 0; index < init_spins.size(); ++index) {
248  if (spin[index] != init_spins[index]) {
249  update_dE_for_binary(index);
250  update_zero_count_and_binary(index);
251  }
252  }
253  }

255  std::stringstream ss;
256  ss << "Unknown vartype detected in " << __func__ << std::endl;
257  throw std::runtime_error(ss.str());
258  }
259

312  void SetUpdateMatrix() {
313  crs_col_.clear();
314  crs_row_.clear();
315  crs_val_.clear();
316  crs_row_.push_back(0);
317  if (vartype_ == cimod::Vartype::SPIN) {
318  crs_sign_p_.clear();
319  for (std::size_t row = 0; row < num_spins; ++row) {
320  std::vector<graph::Index> temp_vec;
321  std::vector<std::vector<FloatType>> temp_vec_val (num_spins);
322  std::vector<std::vector<int8_t*>> temp_vec_sign(num_spins);
323  for (const auto &index: connected_J_term_index_[row]) {
324  auto temp_J_term = poly_value_list_[index];
325  auto temp_sign_p = &spin_sign_poly_[index];
326  for (const auto &col: poly_key_list_[index]) {
327  if (row != col) {
328  temp_vec.push_back(col);
329  temp_vec_val [col].push_back(4*temp_J_term);
330  temp_vec_sign[col].push_back(temp_sign_p);
331  }
332  }
333  }
334  std::sort(temp_vec.begin(), temp_vec.end());
335  temp_vec.erase(std::unique(temp_vec.begin(), temp_vec.end()), temp_vec.end());
336  for (const auto &it: temp_vec) {
337  for (std::size_t i = 0; i < temp_vec_val[it].size(); ++i) {
338  crs_col_.push_back(it);
339  crs_val_.push_back(temp_vec_val[it][i]);
340  crs_sign_p_.push_back(temp_vec_sign[it][i]);
341  }
342  }
343  crs_row_.push_back(crs_col_.size());
344  }
345  }
346  else if (vartype_ == cimod::Vartype::BINARY) {
347  crs_zero_count_p_.clear();
348  for (std::size_t row = 0; row < num_spins; ++row) {
349  std::vector<graph::Index> temp_vec;
350  std::vector<std::vector<FloatType>> temp_vec_val(num_spins);
351  std::vector<std::vector<std::size_t*>> temp_vec_zero_count(num_spins);
352  for (const auto &index: connected_J_term_index_[row]) {
353  auto temp_J_term = poly_value_list_[index];
354  auto temp_zero_count_p = &binary_zero_count_poly_[index];
355  for (const auto &col: poly_key_list_[index]) {
356  if (row != col) {
357  temp_vec.push_back(col);
358  temp_vec_val[col].push_back(temp_J_term);
359  temp_vec_zero_count[col].push_back(temp_zero_count_p);
360  }
361  }
362  }
363  std::sort(temp_vec.begin(), temp_vec.end());
364  temp_vec.erase(std::unique(temp_vec.begin(), temp_vec.end()), temp_vec.end());
365  for (const auto &it: temp_vec) {
366  for (std::size_t i = 0; i < temp_vec_val[it].size(); ++i) {
367  crs_col_.push_back(it);
368  crs_val_.push_back(temp_vec_val[it][i]);
369  crs_zero_count_p_.push_back(temp_vec_zero_count[it][i]);
370  }
371  }
372  crs_row_.push_back(crs_col_.size());
373  }
374  }
375  else {
376  throw std::runtime_error("Unknown vartype detected");
377  }
378  }

380  void SetPolynomialIndex() {
381  connected_J_term_index_.resize(num_spins);
382  if (vartype_ == cimod::Vartype::SPIN) {
383  spin_sign_poly_.resize(num_interactions_);
384  for (std::size_t i = 0; i < num_interactions_; ++i) {
385  graph::Spin spin_multiple = 1;
386  for (const auto &it: poly_key_list_[i]) {
387  spin_multiple *= spin[it];
388  connected_J_term_index_[it].push_back(i);
389  }
390  spin_sign_poly_[i] = static_cast<int8_t>(spin_multiple);
391  }
392  }
393  else if (vartype_ == cimod::Vartype::BINARY) {
394  binary_zero_count_poly_.resize(num_interactions_);
395  for (std::size_t i = 0; i < num_interactions_; ++i) {
396  std::size_t zero_count = 0;
397  graph::Binary binary_multiple = 1;
398  for (const auto &it: poly_key_list_[i]) {
399  binary_multiple *= spin[it];
400  connected_J_term_index_[it].push_back(i);
401  if (spin[it] == 0) {
402  zero_count++;
403  }
404  }
405  binary_zero_count_poly_[i] = zero_count;
406  }
407  }
408  else {
409  throw std::runtime_error("Unknown vartype detected");
410  }

413  void SetPolyKeysAndValues(const cimod::PolynomialKeyList<graph::Index> &input_keys, const cimod::PolynomialValueList<FloatType> &input_values, const bool relabel_flag) {
414  if (input_keys.size() != input_values.size()) {
415  throw std::runtime_error("The sizes of key_list and value_list must match each other");
416  }
417 
418  poly_key_list_ .clear();
419  poly_value_list_.clear();
420 
421  if (relabel_flag) {
422  std::unordered_set<graph::Index> variables;
423  for (std::size_t i = 0; i < input_keys.size(); ++i) {
424  if (input_values[i] != 0.0) {
425  for (const auto &it: input_keys[i]) {
426  variables.emplace(it);
427  }
428  }
429  }
430  std::vector<graph::Index> sorted_variables(variables.begin(), variables.end());
431  std::sort(sorted_variables.begin(), sorted_variables.end());
432 
433  std::unordered_map<graph::Index, graph::Index> relabeld_variables;
434  for (std::size_t i = 0; i < sorted_variables.size(); ++i) {
435  relabeld_variables[sorted_variables[i]] = i;
436  }
437 
438  for (std::size_t i = 0; i < input_keys.size(); ++i) {
439  if (input_values[i] != 0) {
440  std::vector<graph::Index> temp;
441  for (const auto &it: input_keys[i]) {
442  temp.push_back(relabeld_variables[it]);
443  }
444  poly_key_list_.push_back(temp);
445  poly_value_list_.push_back(input_values[i]);
446  }
447  }
448  }
449  else {
450  for (std::size_t i = 0; i < input_keys.size(); ++i) {
451  if (input_values[i] != 0) {
452  poly_key_list_.push_back(input_keys[i]);
453  poly_value_list_.push_back(input_values[i]);
454  }
455  }
456  }
457  num_interactions_ = poly_key_list_.size();
458  }

460  void CheckInitialConditions() const {
461  if (spin.size() != num_spins) {
462  throw std::runtime_error("The number of variables is not equal to the size of the initial spin");
463  }
464  if (poly_key_list_.size() != poly_value_list_.size()) {
465  throw std::runtime_error("The sizes of key_list and value_list must match each other");
466  }
467  if (poly_key_list_.size() <= 0) {
468  throw std::runtime_error("The number of interactions is 0");
469  }
470  if (vartype_ == cimod::Vartype::SPIN) {
471  for (std::size_t i = 0; i < spin.size(); ++i) {
472  if (spin[i] != -1 && spin[i] != 1) {
473  std::stringstream ss;
474  ss << "The variable at " << i << " is " << spin[i] << ".\n";
475  ss << "But the spin variable must be -1 or +1.\n";
476  throw std::runtime_error(ss.str());
477  }
478  }
479  }
480  else if (vartype_ == cimod::Vartype::BINARY) {
481  for (std::size_t i = 0; i < spin.size(); ++i) {
482  if (spin[i] != 0 && spin[i] != 1) {
483  std::stringstream ss;
484  ss << "The variable at " << i << " is " << spin[i] << ".\n";
485  ss << "But the binary variable must be 0 or 1.\n";
486  throw std::runtime_error(ss.str());
487  }
488  }
489  }
490  else {
491  throw std::runtime_error("Unknown vartype detected");
492  }
493  }
