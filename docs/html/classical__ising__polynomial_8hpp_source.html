44  ClassicalIsingPolynomial(const graph::Spins &init_spins, const graph::Polynomial<FloatType> &init_interaction):
45  num_spins(init_interaction.get_num_spins()), spin(init_spins), vartype(init_interaction.GetVartype()) {
46 
47  //Check if the number of the initial spins/binaries is equal to num_spins defined from the Polynomial graph class.
48  assert(init_spins.size() == num_spins);
49 
50  //Check if the initial spin/binary configurations are valid for vartype defined from the Polynomial graph class.
51  CheckVariables();
52 
53  //Set max_variable_, which is the max index of the std::vector representing the initial spin/binary configurations.
54  SetMaxVariable(init_interaction.GetInteractions());
55 
56  //Receive the interactions from the Polynomial graph class.
57  //If the "max_variable_" is larger than or equal to num_spins,
58  //which means that the index of the spin/binary variables does not start at zero, relabel interactions.
59  //"const_cast" is used here to have access to the interactions through operator[]. Note that "init_interaction" should not be changed.
60  auto &&interaction = (GetMaxVariable() < num_spins) ?
61  const_cast<Interactions&>(init_interaction.GetInteractions()) :
62  RelabelInteractions(const_cast<Interactions&>(init_interaction.GetInteractions()));
63 
64  //Set "interacted_spins_", which stores the set of spins/binaries corresponding to the interactions.
65  SetInteractedSpins(interaction);

68  //If "vartype" is cimod::SPIN, "sign_" is also set, else if "vartype" is cimod::BINARY, "zero_count_" is also set.

86  void CheckVariables() {
87  if (vartype == cimod::Vartype::SPIN) {
88  for (std::size_t i = 0; i < spin.size(); ++i) {
89  if (spin[i] != -1 && spin[i] != 1) {
90  std::stringstream ss;
91  ss << "The variable at " << i << " is " << spin[i] << ".\n";
92  ss << "But the spin variable must be -1 or +1.\n";
93  throw std::runtime_error(ss.str());
94  }
95  }
96  }
97  else if (vartype == cimod::Vartype::BINARY) {
98  for (std::size_t i = 0; i < spin.size(); ++i) {
99  if (spin[i] != 0 && spin[i] != 1) {
100  std::stringstream ss;
101  ss << "The variable at " << i << " is " << spin[i] << ".\n";
102  ss << "But the binary variable must be 0 or 1.\n";
103  throw std::runtime_error(ss.str());
104  }
105  }
106  }
107  else {
108  std::stringstream ss;
109  ss << "Unknown vartype detected in " << __func__ << std::endl;
110  throw std::runtime_error(ss.str());
111  }

141  void SetUpdateMatrix(std::vector<std::vector<graph::Index>> &interacted_spins) {
142  crs_col.clear();
143  crs_row.clear();
144  crs_val.clear();
145  crs_row.push_back(0);
146  if (vartype == cimod::Vartype::SPIN) {
147  crs_sign_p.clear();
148  for (graph::Index index_spin_row = 0; index_spin_row < num_spins; ++index_spin_row) {
149  std::vector<graph::Index> temp_vec;
150  std::vector<std::vector<FloatType>> temp_vec_val(num_spins);
151  std::vector<std::vector<int8_t*>> temp_vec_sign(num_spins);
152  for (const auto &index_spin_term: connected_J_term_index_[index_spin_row]) {
153  auto temp_J_term = J_term_[index_spin_term];
154  auto temp_sign_p = &sign_[index_spin_term];
155  for (const auto &index_spin_col: interacted_spins[index_spin_term]) {
156  if (index_spin_row != index_spin_col) {
157  temp_vec.push_back(index_spin_col);
158  temp_vec_val[index_spin_col].push_back(4*temp_J_term);
159  temp_vec_sign[index_spin_col].push_back(temp_sign_p);
160  }
161  }
162  }
163  std::sort(temp_vec.begin(), temp_vec.end());
164  temp_vec.erase(std::unique(temp_vec.begin(), temp_vec.end()), temp_vec.end());
165  for (const auto &it: temp_vec) {
166  for (std::size_t i = 0; i < temp_vec_val[it].size(); ++i) {
167  crs_col.push_back(it);
168  crs_val.push_back(temp_vec_val[it][i]);
169  crs_sign_p.push_back(temp_vec_sign[it][i]);
170  }
171  }
172  crs_row.push_back(crs_col.size());
173  }
174  }
175  else if (vartype == cimod::Vartype::BINARY) {
176  crs_zero_count_p.clear();
177  for (graph::Index index_spin_row = 0; index_spin_row < num_spins; ++index_spin_row) {
178  std::vector<graph::Index> temp_vec;
179  std::vector<std::vector<FloatType>> temp_vec_val(num_spins);
180  std::vector<std::vector<uint64_t*>> temp_vec_zero_count(num_spins);
181  for (const auto &index_spin_term: connected_J_term_index_[index_spin_row]) {
182  auto temp_J_term = J_term_[index_spin_term];
183  auto temp_zero_count_p = &zero_count_[index_spin_term];
184  for (const auto &index_spin_col: interacted_spins[index_spin_term]) {
185  if (index_spin_row != index_spin_col) {
186  temp_vec.push_back(index_spin_col);
187  temp_vec_val[index_spin_col].push_back(temp_J_term);
188  temp_vec_zero_count[index_spin_col].push_back(temp_zero_count_p);
189  }
190  }
191  }
192  std::sort(temp_vec.begin(), temp_vec.end());
193  temp_vec.erase(std::unique(temp_vec.begin(), temp_vec.end()), temp_vec.end());
194  for (const auto &it: temp_vec) {
195  for (std::size_t i = 0; i < temp_vec_val[it].size(); ++i) {
196  crs_col.push_back(it);
197  crs_val.push_back(temp_vec_val[it][i]);
198  crs_zero_count_p.push_back(temp_vec_zero_count[it][i]);
199  }
200  }
201  crs_row.push_back(crs_col.size());
202  }
203  }
204  else {
205  std::stringstream ss;
206  ss << "Unknown vartype detected in " << __func__ << std::endl;
207  throw std::runtime_error(ss.str());
208  }

331  void SetInteractedSpins(const Interactions &interaction) {
332  interacted_spins_.clear();
333  interacted_spins_.reserve(interaction.size());
334  for (const auto &it: interaction) {
335  if (it.first.size() > 0 && std::abs(it.second) > 0.0) {
336  interacted_spins_.push_back(it.first);
337  }
338  }
339 
340  //nameless function to sort "interacted_spins"
341  auto comp = [](const auto &lhs, const auto &rhs) {
342  if (lhs.size() != rhs.size()) {
343  return lhs.size() < rhs.size();
344  }
345  else {
346  if (lhs[0] != rhs[0]) {
347  return lhs[0] < rhs[0];
348  }
349  else {
350  return lhs[1] < rhs[1];
351  }
352  }
353  };
354 
355  //Sort "interacted_spins" for continuous memory access
356  std::sort(interacted_spins_.begin(), interacted_spins_.end(), comp);
357  }

361  void SetJTerm(Interactions &interaction) {
362  connected_J_term_index_.resize(num_spins);
363  J_term_.resize(interacted_spins_.size());
364  if (vartype == cimod::Vartype::SPIN) {
365  sign_.resize(interacted_spins_.size());
366  for (std::size_t i = 0; i < interacted_spins_.size(); ++i) {
367  graph::Spin temp_spin_multiple = 1;
368  if (interacted_spins_[i].size() > 0) {
369  for (const auto &index_spin: interacted_spins_[i]) {
370  temp_spin_multiple *= spin[index_spin];
371  connected_J_term_index_[index_spin].push_back(i);
372  }
373  }
374  J_term_[i] = interaction[interacted_spins_[i]];
375  sign_[i] = static_cast<int8_t>(temp_spin_multiple);
376  }
377  }
378  else if (vartype == cimod::Vartype::BINARY) {
379  zero_count_.resize(interacted_spins_.size());
380  for (std::size_t i = 0; i < interacted_spins_.size(); ++i) {
381  uint64_t temp_zero_count = 0;
382  graph::Binary temp_binary_multiple = 1;
383  if (interacted_spins_[i].size() > 0) {
384  for (const auto &index_binary: interacted_spins_[i]) {
385  temp_binary_multiple *= spin[index_binary];
386  connected_J_term_index_[index_binary].push_back(i);
387  if (spin[index_binary] == 0) {
388  temp_zero_count++;
389  }
390  }
391  }
392  J_term_[i] = interaction[interacted_spins_[i]];
393  zero_count_[i] = temp_zero_count;
394  }
395  }
396  else {
397  std::stringstream ss;
398  ss << "Unknown vartype detected in " << __func__ << std::endl;
399  throw std::runtime_error(ss.str());
400  }
401  }

407  Interactions RelabelInteractions(Interactions &input_interactions) {
408  //Extract variables from the keys of Interactions (std::unordered_map<std::vector<graph::Index>, FloatType, utility::VectorHash>)
409  std::unordered_set<graph::Index> variable_set;
410  for (const auto &it: input_interactions) {
411  for (const auto &index: it.first) {
412  variable_set.emplace(index);
413  }
414  }
415 
416  //Convert std::unordered_set to std::vector
417  std::vector<graph::Index> variables(variable_set.begin(), variable_set.end());
418 
419  //Sort extracted variables
420  std::sort(variables.begin(), variables.end());
421 
422  //Relabel interactions
423  std::unordered_map<graph::Index, graph::Index> relabeld_variables;
424  for (std::size_t i = 0; i < variables.size(); ++i) {
425  relabeld_variables[variables[i]] = i;
426  }
427 
428  Interactions relabeled_interactions;
429 
430  for (const auto &it: input_interactions) {
431  std::vector<graph::Index> temp_vec(it.first.size());
432  for (std::size_t i = 0; i < it.first.size(); ++i) {
433  temp_vec[i] = relabeld_variables[it.first[i]];
434  }
435  relabeled_interactions[temp_vec] = it.second;
436  }
437  return relabeled_interactions;
438  }

441  void SetdE() {
442  dE.resize(num_spins);
443  if (vartype == cimod::Vartype::SPIN) {
444 //#pragma omp parallel for //Maybe OK but afraid so comment out for now
445  for (graph::Index i = 0; i < num_spins; ++i) {
446  FloatType temp_energy = 0.0;
447  for (const auto &it: connected_J_term_index_[i]) {
448  temp_energy += J_term_[it]*sign_[it];
449  }
450  dE[i] = -2*temp_energy;
451  }
452  }
453  else if (vartype == cimod::Vartype::BINARY) {
454 //#pragma omp parallel for //Maybe OK but afraid so comment out for now
455  for (graph::Index i = 0; i < num_spins; ++i) {
456  FloatType temp_energy = 0.0;
457  auto temp_spin = spin[i];
458  for (const auto &it: connected_J_term_index_[i]) {
459  temp_energy += J_term_[it]*Sign(temp_spin)*ZeroOrOne(temp_spin, zero_count_[it]);
460  }
461  dE[i] = temp_energy;
462  }
463  }
464  else {
465  std::stringstream ss;
466  ss << "Unknown vartype detected in " << __func__ << std::endl;
467  throw std::runtime_error(ss.str());
468  }
469  }
