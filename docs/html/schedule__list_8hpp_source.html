120  ClassicalScheduleList make_classical_schedule_list(double beta_min, double beta_max, std::size_t one_mc_step, std::size_t num_call_updater) {
121  double r_beta = std::pow(beta_max/beta_min, 1.0/static_cast<double>(num_call_updater - 1));
122  double beta = beta_min;
123 
124  auto schedule_list = ClassicalScheduleList(num_call_updater);
125  for (auto& schedule : schedule_list) {
126  schedule.one_mc_step = one_mc_step;
127  schedule.updater_parameter = ClassicalUpdaterParameter(beta);
128  beta *= r_beta;
129  }
130 
131  return schedule_list;
132  }

143  TransverseFieldScheduleList make_transverse_field_schedule_list(double beta, std::size_t one_mc_step, std::size_t num_call_updater) {
