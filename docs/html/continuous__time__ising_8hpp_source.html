46  ContinuousTimeIsing(const SpinConfiguration& init_spin_config,
47  const GraphType& init_interaction,

148  size_t get_temporal_spin_index(graph::Index site_index, TimeType time_point) const {
149  static const auto first_lt = [](CutPoint x, CutPoint y) { return x.first < y.first; };
150  // function to compare two time points (lt; less than)
151 
152  const auto& timeline = this->spin_config[site_index];
153  const auto dummy_cut = CutPoint(time_point, 0); // dummy variable for binary search
154  auto found_itr = std::upper_bound(timeline.begin(),
155  timeline.end(),
156  dummy_cut,
157  first_lt);
158 
159  if(found_itr == timeline.begin()) { // if the time_point lies before any time points
160  found_itr = timeline.end() - 1; // periodic boundary condition
161  } else {
162  found_itr--;
163  }
164 
165  return std::distance(timeline.begin(), found_itr);
166  }
