61  template<typename RandomNumberEngine>
62  inline static void update(ClIsing& system,
63  RandomNumberEngine& random_numder_engine,
64  const utility::ClassicalUpdaterParameter& parameter) {
65  // set probability distribution object
66  // to select candidate for flip at random
67  auto uid = std::uniform_int_distribution<std::size_t>(0, system.num_spins-1); //to avoid flipping last spin (must be set to 1.)
68  // to do Metroopolis
69  auto urd = std::uniform_real_distribution<>(0, 1.0);
70 
71  for (std::size_t time = 0; time < system.num_spins; ++time) {
72 
73  // index of spin selected at random
74  const auto index = uid(random_numder_engine);
75 
76  // local energy difference (matrix multiplication)
77  assert(index < system.num_spins);
78  FloatType dE = -2*system.spin(index)*(system.interaction.row(index).dot(system.spin));
79 
80  // Flip the spin?
81  if (dE < 0 || std::exp( -parameter.beta * dE) > urd(random_numder_engine)) {
82  system.spin(index) *= -1;
83  }
84 
85  //assure that the dummy spin is not changed.
86  system.spin(system.num_spins) = 1;
87  }
88  }
89  };

118  template<typename RandomNumberEngine>
119  inline static void update(QIsing& system,
120  RandomNumberEngine& random_numder_engine,
121  const utility::TransverseFieldUpdaterParameter& parameter) {
122 
123  //get number of classical spins
124  std::size_t num_classical_spins = system.num_classical_spins;
125  //get number of trotter slices
126  std::size_t num_trotter_slices = system.trotter_spins.cols();
127 
128  auto uid = std::uniform_int_distribution<std::size_t>{0, num_classical_spins-1};
129  auto uid_trotter = std::uniform_int_distribution<std::size_t>{0, num_trotter_slices-1};
130 
131  //do metropolis
132  auto urd = std::uniform_real_distribution<>(0, 1.0);
133 
134  //aliases
135  auto& spins = system.trotter_spins;
136  auto& gamma = system.gamma;
137  auto& beta = parameter.beta;
138  auto& s = parameter.s;
139 
140  for(std::size_t i=0; i<num_classical_spins*num_trotter_slices; i++){
141  //select random trotter slice
142  std::size_t index_trot = uid_trotter(random_numder_engine);
143  //select random classical spin index
144  std::size_t index = uid(random_numder_engine);
145  //do metropolis
146  FloatType dE = 0;
147  assert(index < num_classical_spins);
148  assert(index_trot < num_trotter_slices);
149  //calculate matrix dot product
150  dE += -2 * s * (beta/num_trotter_slices) * spins(index, index_trot)*(system.interaction.row(index).dot(spins.col(index_trot)));
151 
152  //trotter direction
153  dE += -2 * (1/2.) * log(tanh(beta* gamma * (1.0-s) /num_trotter_slices)) * spins(index, index_trot)*
154  ( spins(index, mod_t((int64_t)index_trot+1, num_trotter_slices))
155  + spins(index, mod_t((int64_t)index_trot-1, num_trotter_slices)));
156 
157  //metropolis
158  if(dE < 0 || exp(-dE) > urd(random_numder_engine)){
159  spins(index, index_trot) *= -1;
160  }
161 
162  }
163  }
164 
165  private:
166  inline static std::size_t mod_t(std::int64_t a, std::size_t num_trotter_slices){
167  //a -> [-1:num_trotter_slices]
168  //return a%num_trotter_slices (a>0), num_trotter_slices-1 (a==-1)
169  return (a+num_trotter_slices)%num_trotter_slices;
170  }
