61  template<typename RandomNumberEngine>
62  inline static FloatType update(ClIsing& system,
63  RandomNumberEngine& random_numder_engine,
64  const utility::ClassicalUpdaterParameter& parameter) {
65  // set probability distribution object
66  // to select candidate for flip at random
67  auto uid = std::uniform_int_distribution<std::size_t>(0, system.spin.size()-1);
68  // to do Metropolis
69  auto urd = std::uniform_real_distribution<>(0, 1.0);
70 
71  // energy difference
72  FloatType total_dE = 0;
73 
74  for (std::size_t time = 0, num_spins = system.spin.size(); time < num_spins; ++time) {
75  // index of spin selected at random
76  const auto index = uid(random_numder_engine);
77  assert(index < num_spins);
78 
79  // local energy difference
80  FloatType dE = 0;
81  for (auto&& adj_index : system.interaction.adj_nodes(index)) {
82  dE += -2.0 * system.spin[index] * (index != adj_index ? (system.interaction.J(index, adj_index) * system.spin[adj_index])
83  : system.interaction.h(index));
84  }
85 
86  // Flip the spin?
87  if (dE < 0 || std::exp( -parameter.beta * dE) > urd(random_numder_engine)) {
88  system.spin[index] *= -1;
89  total_dE += dE;
90  }
91  }
92 
93  return total_dE;
94  }
95  };

126  inline static FloatType update(ClIsing& system,
127  RandomNumberEngine& random_numder_engine,
128  const utility::ClassicalUpdaterParameter& parameter) {
129  // set probability distribution object
130  // to select candidate for flip at random
131  auto uid = std::uniform_int_distribution<std::size_t>(0, system.num_spins-1); //to avoid flipping last spin (must be set to 1.)
132  // to do Metroopolis
133  auto urd = std::uniform_real_distribution<>(0, 1.0);
134 
135  // energy difference
136  FloatType total_dE = 0;
137 
138  for (std::size_t time = 0; time < system.num_spins; ++time) {
139 
140  // index of spin selected at random
141  const auto index = uid(random_numder_engine);
142 
143  // local energy difference (matrix multiplication)
144  assert(index < system.num_spins);
145  FloatType dE = -2*system.spin(index)*(system.interaction.row(index).dot(system.spin));
146 
147  // Flip the spin?
148  if (dE < 0 || std::exp( -parameter.beta * dE) > urd(random_numder_engine)) {
149  system.spin(index) *= -1;
150  total_dE += dE;
151  }
152 
153  //assure that the dummy spin is not changed.
154  system.spin(system.num_spins) = 1;
155  }
156 
157  return total_dE;
158  }

188  template<typename RandomNumberEngine>
189  inline static FloatType update(QIsing& system,
190  RandomNumberEngine& random_numder_engine,
191  const utility::TransverseFieldUpdaterParameter& parameter) {
192 
193  //get number of classical spins
194  std::size_t num_classical_spins = system.trotter_spins[0].size();
195  //get number of trotter slices
196  std::size_t num_trotter_slices = system.trotter_spins.size();
197 
198  auto uid = std::uniform_int_distribution<std::size_t>{0, num_classical_spins-1};
199  auto uid_trotter = std::uniform_int_distribution<std::size_t>{0, num_trotter_slices-1};
200 
201  //do metropolis
202  auto urd = std::uniform_real_distribution<>(0, 1.0);
203 
204  //aliases
205  auto& spins = system.trotter_spins;
206  auto& gamma = system.gamma;
207  auto& beta = parameter.beta;
208  auto& s = parameter.s;
209 
210  FloatType totaldE = 0;
211 
212  for(std::size_t i=0; i<num_classical_spins*num_trotter_slices; i++){
213  //select random trotter slice
214  std::size_t index_trot = uid_trotter(random_numder_engine);
215  //select random classical spin index
216  std::size_t index = uid(random_numder_engine);
217  //do metropolis
218  FloatType dE = 0;
219  //calculate adjacent nodes
220  assert(index < num_classical_spins);
221  assert(index_trot < num_trotter_slices);
222  for(auto&& adj_index : system.interaction.adj_nodes(index)){
223  dE += -2 * s * (beta/num_trotter_slices) * spins[index_trot][index] * (index != adj_index ? (system.interaction.J(index, adj_index) * spins[index_trot][adj_index]) : system.interaction.h(index));
224  }
225 
226  //trotter direction
227  dE += -2 * (1/2.) * log(tanh(beta* gamma * (1.0-s) /num_trotter_slices)) * spins[index_trot][index]*
228  ( spins[mod_t((int64_t)index_trot+1, num_trotter_slices)][index]
229  + spins[mod_t((int64_t)index_trot-1, num_trotter_slices)][index]);
230 
231  //metropolis
232  if(dE < 0 || exp(-dE) > urd(random_numder_engine)){
233  spins[index_trot][index] *= -1;
234  totaldE += dE;
235  }
236 
237  }
238 
239  return totaldE;
240  }
241 
242  private:
243  inline static std::size_t mod_t(std::int64_t a, std::size_t num_trotter_slices){
244  //a -> [-1:num_trotter_slices]
245  //return a%num_trotter_slices (a>0), num_trotter_slices-1 (a==-1)
246  return (a+num_trotter_slices)%num_trotter_slices;
247  }
248  };
