71  Eigen::setNbThreads(1);
72  Eigen::initParallel();

74  // do a iteraction except for the auxiliary spin
75  for (std::size_t index = 0; index < system.num_spins; ++index) {
76 
77  if (system.dE(index) <= 0 || std::exp( -parameter.beta * system.dE(index)) > urd(random_number_engine)) {
78  // update dE
79  system.dE += 4 * system.spin(index) * (system.interaction.row(index).transpose().cwiseProduct(system.spin));
