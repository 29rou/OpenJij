189  FloatType CalculateEnergy(const Spins& spins) const {
190  if(spins.size() != Graph::size()){
191  throw std::out_of_range("Out of range in CalclateEnergy in Polynomial graph.");
192  }
193  FloatType energy = 0.0;
194  for (const auto &it: J_) {
195  Spin temp_spin_multiple = 1;
196  for (const auto &index: it.first) {
197  temp_spin_multiple *= spins[index];
198  }
199  energy += temp_spin_multiple*it.second;
200  }
201  return energy;
202  }
