44  template<typename GraphType>
45  struct SwendsenWang<system::ClassicalIsing<GraphType, false>> {
46 
47  using ClIsing = system::ClassicalIsing<GraphType, false>;
48  using FloatType = typename GraphType::value_type;
49 
50  template<typename RandomNumberEngine>
51  inline static void update(ClIsing& system,
52  RandomNumberEngine& random_number_engine,
53  const utility::ClassicalUpdaterParameter& parameter) {
54  auto urd = std::uniform_real_distribution<>(0, 1.0);
55  const auto num_spin = system.spin.size();
56 
57  // 1. update bonds
58  auto union_find_tree = utility::UnionFind(num_spin);
59  for (std::size_t node = 0; node < num_spin; ++node) {
60  for (auto&& adj_node : system.interaction.adj_nodes(node)) {
61  if (node >= adj_node) continue;
62  //check if bond can be connected
63  if (system.interaction.J(node, adj_node) * system.spin[node] * system.spin[adj_node] > 0) continue;
64  const auto unite_rate = std::max(static_cast<FloatType>(0.0), static_cast<FloatType>(1.0 - std::exp( - 2.0 * parameter.beta * std::abs(system.interaction.J(node, adj_node)))));
65  if (urd(random_number_engine) < unite_rate)
66  union_find_tree.unite_sets(node, adj_node);
67  }
68  }
69 
70  // 2. make clusters
71  const auto cluster_map = [num_spin, &union_find_tree](){
72  auto cluster_map = std::unordered_multimap<utility::UnionFind::Node, utility::UnionFind::Node>();
73  for (std::size_t node = 0; node < num_spin; ++node) {
74  cluster_map.insert({union_find_tree.find_set(node), node});
75  }
76  return cluster_map;
77  }();
78 
79  // 3. update spin states in each cluster
80  for (auto&& c : union_find_tree.get_roots()) {
81  const auto range = cluster_map.equal_range(c);
82 
83  // 3.1. calculate energy \sum_{i \in C} h_i
84  double energy_magnetic = 0.0;
85  for (auto itr = range.first, last = range.second; itr != last; ++itr) {
86  const auto idx = itr->second;
87  energy_magnetic += system.interaction.h(idx)*system.spin[idx];
88  }
89 
90  // 3.2. decide spin state
91  const FloatType probability = 1.0 / ( std::exp(-2 * parameter.beta * energy_magnetic) + 1.0 );
92  if(urd(random_number_engine) < probability){
93  // 3.3. update spin states
94  for (auto itr = range.first, last = range.second; itr != last; ++itr) {
95  const auto idx = itr->second;
96  system.spin[idx] *= -1;
97  }
98  }
99  }
100 
101  return;
102  }
103  };

118  const utility::ClassicalUpdaterParameter& parameter) {
119  auto urd = std::uniform_real_distribution<>(0, 1.0);
120 
121  // num_spin = system size + additional spin
122  const size_t num_spin = system.spin.size();
123 
124  // 1. update bonds
125  auto union_find_tree = utility::UnionFind(num_spin);
126  for (std::size_t node = 0; node < num_spin; ++node) {
127  for (typename ClIsing::SparseMatrixXx::InnerIterator it(system.interaction, node); it; ++it) {
128  //fetch adjacent node
129  std::size_t adj_node = it.index();
130  //fetch system.interaction(node, adj_node)
131  const FloatType& J = it.value();
132  if (node >= adj_node) continue;
133  //check if bond can be connected
134  if (J * system.spin(node) * system.spin(adj_node) > 0) continue;
135  const auto unite_rate = std::max(static_cast<FloatType>(0.0), static_cast<FloatType>(1.0 - std::exp( - 2.0 * parameter.beta * std::abs(J))));
136  if (urd(random_number_engine) < unite_rate)
137  union_find_tree.unite_sets(node, adj_node);
138  }
139  }

144  for (std::size_t node = 0; node < num_spin; ++node) {
145  cluster_map.insert({union_find_tree.find_set(node), node});
146  }
