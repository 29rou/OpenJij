93  template<typename GraphType>
94  const graph::Spins get_solution(const system::TransverseIsing<GraphType, true>& system){
95  std::size_t minimum_trotter = 0;
96  //aliases
97  auto& spins = system.trotter_spins;
98  double energy = 0.0;
99  double min_energy = std::numeric_limits<double>::max();
100  //get number of trotter slices
101  std::size_t num_trotter_slices = system.trotter_spins.cols();
102  for (std::size_t t=0; t<num_trotter_slices; t++){
103  // calculate classical energy in each classical spin
104  energy = spins.col(t).transpose() * system.interaction * spins.col(t);
105  if(energy < min_energy){
106  minimum_trotter = t;
107  min_energy = energy;
108  }
109  }
110 
111  //convert from Eigen::Vector to std::vector
112  graph::Spins ret_spins(system.num_classical_spins);
113  for(std::size_t i=0; i<system.num_classical_spins; i++){
114  ret_spins[i] = spins(i, minimum_trotter);
115  }
116  return ret_spins;
