145  Square(std::size_t num_row, std::size_t num_column, FloatType init_val=0)
146  : Sparse<FloatType>(num_row*num_column, 4+1), _init_val(init_val), _num_row(num_row), _num_column(num_column){
147  assert(num_row >= 1);
148  assert(num_column >= 1);
149 
150  for(std::size_t r=0; r<_num_row; r++){
151  for(std::size_t c=0; c<_num_column; c++){
152  //open boundary
153  if(r > 0){
154  //MINUS_R
155  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r-1,c)) = _init_val;
156  }
157  if(c > 0){
158  //MINUS_C
159  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r,c-1)) = _init_val;
160  }
161  if(r < num_row-1){
162  //PLUS_R
163  this->Sparse<FloatType>::J(to_ind(r,c), to_ind(r+1,c)) = _init_val;
164  }
165  if(c < num_column-1){
166  //PLUS_C

168  }
169  //local field

171  }
