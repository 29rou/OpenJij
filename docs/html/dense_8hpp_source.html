167  FloatType calc_energy(const Spins& spins) const{
168  if(!(spins.size() == get_num_spins())){
169  throw std::runtime_error("invalid length of spins in Dense");
170  }
171  FloatType ret = 0;
172  for(std::size_t ind=0; ind<this->get_num_spins(); ind++){
173  for(auto& adj_ind : _list_adj_nodes[ind]){
174  if(ind != adj_ind)
175  ret += (1./2) * this->J(ind, adj_ind) * spins[ind] * spins[adj_ind];
176  else
177  ret += this->h(ind) * spins[ind];
178  }
179  }
180 
181  return ret;
182  }
