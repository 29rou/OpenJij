105  void set_interaction_matrix(const Interactions& interaction){
106  if(interaction.rows() != interaction.cols()){
107  std::runtime_error("interaction.rows() != interaction.cols()");
108  }
109 
110  if((size_t)interaction.rows() != get_num_spins() + 1){
111  throw std::runtime_error("invalid matrix size.");
112  }
113 
114  //check if diagonal elements are zero
115  for(size_t i=0; i<(size_t)(interaction.rows()-1); i++){
116  if(interaction(i,i) != 0){
117  throw std::runtime_error("The diagonal elements of interaction matrix must be zero.");
118  }
119  }
120 
121  if(interaction(interaction.rows()-1,interaction.rows()-1) != 1){
122  throw std::runtime_error("The right bottom element of interaction matrix must be unity.");
123  }
124 
125  _J = interaction.template selfadjointView<Eigen::Upper>();
126  }

146  FloatType calc_energy(const Spins& spins) const{
147  if(spins.size() != this->get_num_spins()){
148  std::out_of_range("Out of range in calc_energy in Dense graph.");
149  }
150 
151  using Vec = Eigen::Matrix<FloatType, Eigen::Dynamic, 1, Eigen::ColMajor>;
152  Vec s(get_num_spins()+1);
153  for(size_t i=0; i<spins.size(); i++){
154  s(i) = spins[i];
155  }
156  s(get_num_spins()) = 1;
157 
158  // the energy must be consistent with BinaryQuadraticModel.
159  return (s.transpose()*(_J.template triangularView<Eigen::Upper>()*s))(0,0)-1;
160  }
