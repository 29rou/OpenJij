77  inline bool set_adj_node(Index from, Index to){
78  assert(from < this->get_num_spins());
79  assert(to < this->get_num_spins());
80 
81  //get adjacent nodes of node "from"
82  Nodes& nodes = _list_adj_nodes[from];
83  //check if the node "to" exists in the nodes
84  if(std::find(nodes.begin(), nodes.end(), to)==nodes.end()){
85  //nodes size must be smaller than num_edges

93  //add node
94  nodes.push_back(to);
95  //add node from "to" to "from"
96  set_adj_node(to, from);
97  }
98  return true;

196  FloatType calc_energy(const Spins& spins) const{
197  if(!(spins.size() == get_num_spins())){
198  throw std::runtime_error("invalid length of spins in Sparse");
199  }
200  FloatType ret = 0;
201  for(std::size_t ind=0; ind<this->get_num_spins(); ind++){
202  for(auto& adj_ind : _list_adj_nodes[ind]){
203  if(ind != adj_ind)
204  ret += (1./2) * this->J(ind, adj_ind) * spins[ind] * spins[adj_ind];
205  else
206  ret += this->h(ind) * spins[ind];
207  }
208  }
209 
210  return ret;
211  }
